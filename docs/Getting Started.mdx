```bash
npm i @rebasejs/rebase
```

## Usage

```jsx
import React from "react"
import { Box, Heading, Button } from "@rebasejs/rebase"

export default () => (
  <Box>
    <Heading>Hello</Heading>
    <Button>Rebase</Button>
  </Box>
)
```
